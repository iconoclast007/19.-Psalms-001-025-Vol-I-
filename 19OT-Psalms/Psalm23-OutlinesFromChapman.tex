\begin{compactenum}[I.]
    \item The \textbf{Shepherd's Person} (1)
    \begin{compactenum}[A.]
        \item It has Exalting Comparisons \index[scripture]{Psalms!Psa 023:01a}(Psa 23:1a)\footnote{Jimmy Chapman, 7 January 2021, Just Baptist Outlines Group}

    \begin{compactenum}[A.]

        \item That includes Still Waters \index[scripture]{Psalms!Psa 023:02b}(Psa 23:2b)
    \end{compactenum}
    \item The \textbf{Shepherd's Paths} (3)
     \begin{compactenum}[A.]
        \item It Emphasizes Roaming \index[scripture]{Psalms!Psa 023:03a}(Psa 23:3a)

    \end{compactenum}
   \item The \textbf{Shepherd's Presence} (4)
     \begin{compactenum}[A.]
        \item End Rebellion, Enter Righteousness \index[scripture]{Psalms!Psa 023:04a}(Psa 23:4a)
        \item Enter Righteousness, Enjoy Reality \index[scripture]{Psalms!Psa 023:04b}(Psa 23:4b)
    \end{compactenum}
    \item The \textbf{Shepherd's Preparation} (5)
    \begin{compactenum}[A.]
        \item That Included Blessed Arrangments \index[scripture]{Psalms!Psa 023:05a}(Psa 23:5a)
        \item That Included Blessed Anointing \index[scripture]{Psalms!Psa 023:05b}(Psa 23:5b)
    \end{compactenum}
    \item The \textbf{Shepherd's Peace} (6)
    \begin{compactenum}[A.]
        \item Sheep Enjoyed Guaranteed Peace \index[scripture]{Psalms!Psa 023:06a}(Psa 23:6a) \item Sheep Enjoyed Glorious Peace \index[scripture]{Psalms!Psa 023:06b}(Psa 23:6b)
    \end{compactenum}
\end{compactenum}
